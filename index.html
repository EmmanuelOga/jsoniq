Pending Update Lists

var store = new MemoryStore();

var obj = { a: 1, b: { c: 1 } };
var id = store.put(obj);
store
.insertIntoObject(id, ['b'], { d: 1 })
.deleteFromObject(id, ['b'], ['c']);

store.commit();

assert.equal(obj.b.d, 1);
assert.equal(obj.b.c, undefined);
